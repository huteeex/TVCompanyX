        end note

        end note

            [*] --> АнализЗаявки

            ПроверкаРасписания --> ПроверкаТребований
            ПроверкаТребований --> ПринятиеРешения
        }
    }
    
    ReadyForReview --> Approved : одобрить

    }

    }

    state Completed {
        [*] --> АрхивированиеДанных
        
        note right of АрхивированиеДанных
            entry / переместить в архив
            do / хранение истории
        end note
    }
    
    Completed --> [*]
    
    state Rejected {
        [*] --> УказаниеПричины
        УказаниеПричины --> УведомлениеОбОтклонении
        
        note left of Rejected
            entry / сохранить причину
            entry / уведомить агента
            entry / уведомить клиента
        end note
    }
    
    Rejected --> InProgress : вернуть агенту
    Rejected --> Cancelled : отменить
    
    state Cancelled {
        [*] --> ЗакрытиеЗаявки
        
        note left of Cancelled
            entry / архивировать
            entry / закрыть процессы
        end note
    }
    
    Cancelled --> [*]
    
    note right of Pending

    end note

## 📊 Описание состояний рекламной заявки

### 🟢 Основные состояния

1. **Создание** - начальное составное состояние
   - Ввод данных (шоу, дата, длительность)
   - Расчёт стоимости
   - Проверка корректности
   - Готовность к отправке

2. **Pending** (ожидание)
   - Заявка сохранена в БД
   - Ожидает назначения рекламного агента
   - Коммерческий отдел получил уведомление

3. **InProgress** (в работе)
   - Агент назначен на заявку
   - Открытие чата с клиентом
   - Обмен сообщениями
   - Согласование деталей размещения
   - Подготовка материалов

4. **ReadyForReview** (готова к проверке)
   - Агент передал заявку в коммерцию
   - Ожидание проверки коммерческим отделом
   - Анализ заявки
   - Проверка расписания эфира
   - Проверка соответствия требованиям
   - Принятие решения

### ✅ Путь успешного выполнения

5. **Approved** (одобрена)
   - Заявка одобрена коммерческим отделом
   - Установлена дата одобрения (approved_at)
   - Рассчитана комиссия агента (5% от стоимости)
   - Уведомления отправлены всем участникам

6. **ContractSent** (договор отправлен)
   - Сгенерирован уникальный номер договора
   - Договор сохранён в БД
   - Клиент получил уведомление

7. **ContractViewed** (договор просмотрен)
   - Клиент открыл и изучил договор
   - Установлена отметка viewed_at

8. **ContractDownloaded** (договор скачан)
   - Клиент скачал PDF файл договора
   - Установлена отметка downloaded_at
   - Ожидание оплаты вне системы

9. **Paid** (оплачена)
   - Бухгалтер зарегистрировал платёж
   - Проверка совпадения суммы
   - Сохранение данных платежа
   - Установлена дата оплаты (payment_date)
   - Обновлена финальная комиссия агента
   - Уведомления всем участникам

10. **Completed** (завершена)
    - Заявка полностью выполнена
    - Данные архивированы
    - Хранение истории

### ❌ Путь отклонения

11. **Rejected** (отклонена)
    - Коммерческий отдел отклонил заявку
    - Указана причина отклонения
    - Уведомления агенту и клиенту
    - Возможность вернуть в работу для исправления

12. **Cancelled** (отменена)
    - Заявка окончательно отменена
    - Закрыты все связанные процессы
    - Архивирование данных

## 🔄 Переходы между состояниями

### Прямой путь (успешное выполнение):
```
Создание → Pending → InProgress → ReadyForReview → Approved → 
ContractSent → ContractViewed → ContractDownloaded → Paid → Completed
```

### Путь с отклонением и возвратом:
```
ReadyForReview → Rejected → InProgress (исправление) → ReadyForReview
```

### Путь с окончательной отменой:
```
ReadyForReview → Rejected → Cancelled
```

## 🎯 События (триггеры переходов)

- **отправить заявку** - клиент подтверждает создание
- **назначить агента** - коммерция назначает ответственного
- **передать в коммерцию** - агент готов к проверке
- **одобрить** - коммерция одобряет заявку
- **отклонить** - коммерция отклоняет заявку
- **создать договор** - агент инициирует создание договора
- **просмотрел договор** - клиент открыл документ
- **скачал договор** - клиент сохранил PDF
- **зарегистрировать платеж** - бухгалтер подтвердил оплату
- **вернуть агенту** - повторная обработка
- **отменить** - окончательная отмена

## 📝 Действия в состояниях

### Entry actions (при входе в состояние):
- Сохранение данных в БД
- Установка временных меток
- Расчёт комиссии
- Отправка уведомлений

### Do actions (во время нахождения в состоянии):
- Ожидание действий пользователя
- Работа с чатом
- Хранение данных

### Exit actions (при выходе из состояния):
- Сохранение финальных данных
- Подготовка данных для следующего состояния
- Закрытие соединений
