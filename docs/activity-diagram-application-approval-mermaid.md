stateDiagram-v2
    direction LR
    
    [*] --> ВходВСистему: Заказчик
    
    state "Дорожка: Заказчик" as CustomerLane {
        ВходВСистему --> ОткрытьФорму: войти
        ОткрытьФорму --> ВыборШоу: открыть раздел
        ВыборШоу --> УказатьДату: выбрать шоу
        УказатьДату --> УказатьДлительность: указать дату
    }
    
    state "Дорожка: Клиентская часть" as ClientLane {
        УказатьДлительность --> ОтправитьЗапрос: указать длительность
    }
    
    state "Дорожка: Серверная часть" as ServerLane {
        ОтправитьЗапрос --> ПолучитьПараметры: отправить
        ПолучитьПараметры --> НайтиШоу: получить
        НайтиШоу --> ПолучитьЦену: найти в БД
        ПолучитьЦену --> РассчитатьСтоимость: получить цену
        РассчитатьСтоимость --> ВернутьРасчёт: рассчитать
    }
    
    state "Дорожка: Клиентская часть" as ClientLane2 {
        ВернутьРасчёт --> ОтобразитьСтоимость: вернуть
    }
    
    state "Дорожка: Заказчик" as CustomerLane2 {
        ОтобразитьСтоимость --> ПросмотрРасчёта: отобразить
        ПросмотрРасчёта --> РешениеПоЦене: просмотреть
    }
    
    state РешениеПоЦене <<choice>>
    РешениеПоЦене --> ЗаполнитьОписание: согласен
    РешениеПоЦене --> ЗакрытьФорму: не согласен
    
    state "Дорожка: Заказчик" as CustomerLane3 {
        ЗаполнитьОписание --> УказатьТелефон: заполнить
        УказатьТелефон --> ПодтвердитьСоздание: указать
    }
    
    state "Дорожка: Клиентская часть" as ClientLane3 {
        ПодтвердитьСоздание --> ОтправитьЗаявку: подтвердить
    }
    
    state "Дорожка: Серверная часть" as ServerLane2 {
        ОтправитьЗаявку --> ПолучитьДанные: отправить
        ПолучитьДанные --> СохранитьВБД: получить
        
        state fork1 <<fork>>
        СохранитьВБД --> fork1
        fork1 --> СохранитьЗаявку
        fork1 --> СоздатьУведомление
        
        state join1 <<join>>
        СохранитьЗаявку --> join1
        СоздатьУведомление --> join1
        
        join1 --> ОтправитьУведомлениеWS: сохранить
    }
    
    state "Дорожка: Коммерческий отдел" as CommercialLane {
        ОтправитьУведомлениеWS --> ПолучитьУведомление: отправить
        ПолучитьУведомление --> ОткрытьСписок: получить
        ОткрытьСписок --> ПросмотрДеталей: открыть
        ПросмотрДеталей --> НазначитьАгента: просмотреть
    }
    
    state "Дорожка: Серверная часть" as ServerLane3 {
        НазначитьАгента --> ОбновитьСтатус: назначить
        
        state fork2 <<fork>>
        ОбновитьСтатус --> fork2
        fork2 --> СохранитьНазначение
        fork2 --> СоздатьУведомлениеАгенту
        
        state join2 <<join>>
        СохранитьНазначение --> join2
        СоздатьУведомлениеАгенту --> join2
    }
    
    state "Дорожка: Рекламный агент" as AgentLane {
        join2 --> ПолучитьУведомлениеАгент: сохранить
        ПолучитьУведомлениеАгент --> ОткрытьЗаявку: получить
        ОткрытьЗаявку --> СвязатьсяСКлиентом: открыть
        СвязатьсяСКлиентом --> УточнитьДетали: связаться
    }
    
    state "Дорожка: Коммерческий отдел" as CommercialLane2 {
        УточнитьДетали --> ПровестиАнализ: уточнить
        ПровестиАнализ --> РешениеПоЗаявке: провести
    }
    
    state РешениеПоЗаявке <<choice>>
    РешениеПоЗаявке --> Одобрить: соответствует
    РешениеПоЗаявке --> Отклонить: не соответствует
    
    state "Дорожка: Серверная часть" as ServerLane4 {
        Одобрить --> ОбновитьСтатусОдобрена: одобрить
        
        state fork3 <<fork>>
        ОбновитьСтатусОдобрена --> fork3
        fork3 --> РассчитатьКомиссию
        fork3 --> СоздатьЗаписьКомиссии
        fork3 --> ОтправитьУведомленияВсем
        
        state join3 <<join>>
        РассчитатьКомиссию --> join3
        СоздатьЗаписьКомиссии --> join3
        ОтправитьУведомленияВсем --> join3
        
        join3 --> ЗаявкаОдобрена: обработать
        ЗаявкаОдобрена --> [*]: завершить
    }
    
    state "Дорожка: Серверная часть" as ServerLane5 {
        Отклонить --> ОбновитьСтатусОтклонена: отклонить
        
        state fork4 <<fork>>
        ОбновитьСтатусОтклонена --> fork4
        fork4 --> СохранитьПричину
        fork4 --> ОтправитьУведомленияОтклонение
        
        state join4 <<join>>
        СохранитьПричину --> join4
        ОтправитьУведомленияОтклонение --> join4
        
        join4 --> ЗаявкаОтклонена: обработать
        ЗаявкаОтклонена --> [*]: завершить
    }
    
    ЗакрытьФорму --> СозданиеОтменено: закрыть
    СозданиеОтменено --> [*]: отменить
